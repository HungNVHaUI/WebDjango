tạo pr: django-admin startproject mysite
run Server: python manage.py runserver
tạo bảng: python manage.py migrate  //tạo tên bảng trước
tạo model: python manage.py makemigrations "name app"
tạo tài khoản admin: python manage.py createsuperuser
môi trường ảo chạy file Scripts\activate.bat
lấy các thư việ trong môi trường ảo:pip freeze > requirements.txt
cài đặt:pip install -r requirements.txt
thoát venv deactivate
